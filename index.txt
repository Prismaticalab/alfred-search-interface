    <script>
        function startSearch() {
            document.getElementById('startState').style.display = 'none';
            document.getElementById('searchState').classList.add('active');
        }

        function goBack() {
            document.getElementById('searchState').classList.remove('active');
            document.getElementById('startState').style.display = 'block';
            resetSearch();
        }

        function resetSearch() {
            document.getElementById('location').value = '';
            document.getElementById('additional-notes').value = '';
            
            const radios = document.querySelectorAll('input[type="radio"]');
            radios.forEach(radio => radio.checked = false);
            
            updateSummary();
        }

        function updateSummary() {
            const whoSegment = document.querySelector('input[name="who-segment"]:checked');
            const location = document.getElementById('location').value;
            const additionalNotes = document.getElementById('additional-notes').value;
            const quantity = document.querySelector('input[name="quantity"]:checked');
            
            let summaryText = '';
            
            if (whoSegment || location || additionalNotes || quantity) {
                summaryText = '<p style="margin: 0 0 1rem 0;"><strong>Alfred will search for:</strong></p>';
                
                if (whoSegment) {
                    const segmentLabel = whoSegment.nextElementSibling.textContent;
                    summaryText += `<p style="margin: 0 0 0.5rem 0;">‚Ä¢ <strong>Target:</strong> ${segmentLabel}</p>`;
                }
                
                if (location) {
                    summaryText += `<p style="margin: 0 0 0.5rem 0;">‚Ä¢ <strong>Location:</strong> ${location}</p>`;
                }
                
                if (quantity) {
                    const quantityLabel = quantity.value === 'unlimited' ? 'As many as possible' : quantity.value;
                    summaryText += `<p style="margin: 0 0 0.5rem 0;">‚Ä¢ <strong>Quantity:</strong> ${quantityLabel}</p>`;
                }
                
                if (additionalNotes) {
                    summaryText += `<p style="margin: 0 0 0.5rem 0;">‚Ä¢ <strong>Special requirements:</strong> ${additionalNotes}</p>`;
                }
            } else {
                summaryText = '<p style="margin: 0; font-style: italic; color: #9ca3af;">Fill out the fields above to see your search summary...</p>';
            }
            
            document.getElementById('search-summary').innerHTML = summaryText;
        }

        function submitSearch() {
            // Get the current search parameters
            const whoSegment = document.querySelector('input[name="who-segment"]:checked');
            const location = document.getElementById('location').value;
            const additionalNotes = document.getElementById('additional-notes').value;
            const quantity = document.querySelector('input[name="quantity"]:checked');
            
            // Build final summary
            let finalSummary = '';
            
            if (whoSegment) {
                const segmentLabel = whoSegment.nextElementSibling.textContent;
                finalSummary += `<p style="margin: 0 0 0.5rem 0;">‚Ä¢ <strong>Target:</strong> ${segmentLabel}</p>`;
            }
            
            if (location) {
                finalSummary += `<p style="margin: 0 0 0.5rem 0;">‚Ä¢ <strong>Location:</strong> ${location}</p>`;
            }
            
            if (quantity) {
                const quantityLabel = quantity.value === 'unlimited' ? 'As many as possible' : quantity.value;
                finalSummary += `<p style="margin: 0 0 0.5rem 0;">‚Ä¢ <strong>Quantity:</strong> ${quantityLabel}</p>`;
            }
            
            if (additionalNotes) {
                finalSummary += `<p style="margin: 0 0 0.5rem 0;">‚Ä¢ <strong>Special requirements:</strong> ${additionalNotes}</p>`;
            }
            
            if (!finalSummary) {
                finalSummary = '<p style="margin: 0; color: #9ca3af; font-style: italic;">No specific parameters set</p>';
            }
            
            document.getElementById('final-summary').innerHTML = finalSummary;
            
            // Transition to confirmation state
            document.getElementById('searchState').classList.remove('active');
            document.getElementById('confirmationState').classList.add('active');
        }

        function startNewSearch() {
            // Reset form and go back to start state
            resetSearch();
            document.getElementById('confirmationState').classList.remove('active');
            document.getElementById('startState').style.display = 'block';
        }

        function viewDashboard() {
            alert('Redirecting to dashboard... (This would normally take you back to the main CRM)');
        }

        // Add event listeners for real-time summary updates
        document.addEventListener('DOMContentLoaded', function() {
            document.getElementById('location').addEventListener('input', updateSummary);
            document.getElementById('additional-notes').addEventListener('input', updateSummary);
            
            const whoRadios = document.querySelectorAll('input[name="who-segment"]');
            whoRadios.forEach(radio => {
                radio.addEventListener('change', updateSummary);
            });

            const quantityRadios = document.querySelectorAll('input[name="quantity"]');
            quantityRadios.forEach(radio => {
                radio.addEventListener('change', updateSummary);
            });
        });
    </script>
